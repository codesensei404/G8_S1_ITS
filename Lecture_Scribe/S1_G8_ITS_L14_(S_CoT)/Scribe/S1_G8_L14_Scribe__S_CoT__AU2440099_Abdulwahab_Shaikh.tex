\section*{Q1. Random Point on a Line Segment}

\subsection*{Interpretation}

A point is chosen \textbf{uniformly at random} on a line segment of length $(L)$.

Let:
\[
0 < x < L
\]
be the distance from one endpoint.

Two segments formed:

\begin{itemize}
\item $(x)$
\item $(L-x)$
\end{itemize}

Define the ratio:
\[
R = \frac{\min(x,L-x)}{\max(x,L-x)}
\]

We want:
\[
P(R < 1/4)
\]

\subsection*{Step 1: Logical Case Division at the Midpoint}

\subsubsection*{Case 1: $(x \le L/2)$}

Shorter = $(x)$ \\
Longer = $(L-x)$

Condition:
\[
\frac{x}{L-x} < \frac14
\]

\[
4x < L-x
\]

\[
5x < L
\]

\[
x < \frac{L}{5}
\]

\subsubsection*{Case 2: $(x > L/2)$}

Shorter = $(L-x)$ \\
Longer = $(x)$

\[
\frac{L-x}{x} < \frac14
\]

\[
4L - 4x < x
\]

\[
5x > 4L
\]

\[
x > \frac{4L}{5}
\]

\subsection*{Step 2: Determine Valid Region}

\[
(0,L/5) \cup (4L/5,L)
\]

Length of region:
\[
\left(\frac{L}{5}-0\right) + \left(L-\frac{4L}{5}\right)
\]

\[
\frac{L}{5} + \frac{L}{5}
\]

\[
\frac{2L}{5}
\]

\[
P = \frac{2L/5}{L} = \frac{2}{5}
\]

\section*{Q2. Bayesian Decision Rule with Normal Distributions}

\subsection*{Given}

White region:
\[
X|W \sim N(4,4)
\]

\[
\sigma_1 = 2
\]

Black region:
\[
X|B \sim N(6,9)
\]

\[
\sigma_2 = 3
\]

Priors:
\[
P(B)=\alpha,\quad P(W)=1-\alpha
\]

Observation:
\[
X=5
\]

Requirement:
\[
P(B|X=5)=P(W|X=5)=\frac12
\]

\subsection*{Step 1: Apply Bayes’ Theorem}

\[
P(B|X=5)
=
\frac{\alpha f(5|B)}
{\alpha f(5|B)+(1-\alpha)f(5|W)}
=
\frac12
\]

\subsection*{Step 2: Compute Likelihoods}

Normal density:
\[
f(x)=\frac{1}{\sigma\sqrt{2\pi}}
e^{-\frac{(x-\mu)^2}{2\sigma^2}}
\]

For Black:
\[
f(5|B)=\frac{1}{3\sqrt{2\pi}}e^{-1/18}
\]

For White:
\[
f(5|W)=\frac{1}{2\sqrt{2\pi}}e^{-1/8}
\]

Substitute:
\[
\frac{3\alpha}
{3\alpha+2(1-\alpha)e^{-5/72}}
=
\frac12
\]

\subsection*{Step 3: Solve for $(\alpha)$}

\[
6\alpha=3\alpha+2(1-\alpha)(0.93)
\]

\[
3\alpha=1.86-1.86\alpha
\]

\[
4.86\alpha=1.86
\]

\[
\alpha \approx 0.3827
\]

\section*{Q3. Fire Station Location}

\subsection*{(a) Uniform Distribution on $(0,A)$}

\[
X\sim Uniform(0,A)
\]

\[
f_X(x)=\frac1A
\]

Minimize:
\[
E[|X-a|]
\]

\[
E[|X-a|]
=
\int_0^a (a-x)\frac1A dx
+
\int_a^A (x-a)\frac1A dx
\]

\[
E[|X-a|]
=
\frac{2a^2-2aA+A^2}{2A}
\]

\[
\frac{d}{da}
=
\frac1{2A}(4a-2A)
\]

\[
4a=2A
\]

\[
a=\frac A2
\]

\subsection*{(b) Exponential Distribution}

\[
X\sim Exponential(\lambda)
\]

\[
f(x)=\lambda e^{-\lambda x}
\]

\[
E[|X-a|]
=
\int_0^a (a-x)\lambda e^{-\lambda x}dx
+
\int_a^\infty (x-a)\lambda e^{-\lambda x}dx
\]

\[
\frac{d}{da}E[|X-a|]
=
1-2e^{-\lambda a}
\]

\[
2e^{-\lambda a}=1
\]

\[
e^{-\lambda a}=\frac12
\]

\[
a=\frac{\ln2}{\lambda}
\]

\section*{Q5. Conditional Distribution of Poisson Variables}

\[
X\sim Poisson(\lambda_1)
\]

\[
Y\sim Poisson(\lambda_2)
\]

\[
P(X=k|X+Y=n)
=
\frac{
\left(\frac{e^{-\lambda_1}\lambda_1^k}{k!}\right)
\left(\frac{e^{-\lambda_2}\lambda_2^{n-k}}{(n-k)!}\right)
}
{
\frac{e^{-(\lambda_1+\lambda_2)}(\lambda_1+\lambda_2)^n}{n!}
}
\]

\[
=
\binom{n}{k}
\left(\frac{\lambda_1}{\lambda_1+\lambda_2}\right)^k
\left(\frac{\lambda_2}{\lambda_1+\lambda_2}\right)^{n-k}
\]

\section*{Q10. Joint Distribution of $(M_n)$ and $(M_{n+1})$}

\[
M_n=\max(X_1,\dots,X_n)
\]

\[
P(M_n\le x)=F(x)^n
\]

Case 1: $(x\le y)$

\[
P(M_n\le x,M_{n+1}\le y)
=
F(x)^nF(y)
\]

Case 2: $(x>y)$

\[
P(M_n\le x,M_{n+1}\le y)
=
F(y)^{n+1}
\]
